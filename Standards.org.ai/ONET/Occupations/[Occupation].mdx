export const items = CSV.fetch('https://www.onetcenter.org/dl_files/database/db_30_0_text/Occupation%20Data.txt')
  .then(rows => rows
      .filter(row => row['Title'] && row['O*NET-SOC Code'])
      .map(row => {
      const title = row['Title'];
      const code = row['O*NET-SOC Code'];
      const description = row['Description'];
      // PascalCase for Component/File naming
      const slug = _.upperFirst(_.camelCase(title));
      
      return {
          id: slug, // Filename
          slug: slug,
          code: code,
          title: title,
          description: description,
          $id: `https://graph.org.ai/occupations/${slug}`
      };
  }));
