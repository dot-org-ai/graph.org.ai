export const items = CSV.fetch('https://www.onetcenter.org/dl_files/database/db_30_0_text/Tools%20Used.txt')
  .then(rows => {
      const unique = _.uniqBy(rows, 'Commodity Title');
      
      return unique
        .filter(row => row['Commodity Title'])
        .map(row => {
          const name = row['Commodity Title'];
          const slug = _.upperFirst(_.camelCase(name));
          
          return {
              id: slug,
              slug: slug,
              name: name,
              description: `Tool class: ${name}`,
              $id: `https://graph.org.ai/tools/${slug}`
          };
      });
  });
