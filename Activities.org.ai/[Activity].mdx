  .then(rows => {
      return rows.map(row => {
          const dwaId = row['DWA ID'];
          const title = row['DWA Title'];

          return {
              id: dwaId,
              slug: dwaId,
              dwaId: dwaId,
              title: title,
              dwaTitle: title,
              iwaId: row['IWA ID'],
              elementId: row['Element ID'],
              $id: `https://graph.org.ai/activities/${dwaId}`
          };
      });
  });
