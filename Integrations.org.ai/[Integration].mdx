export const items = fetch('https://zapier.com/api/v4/services')
  .then(res => res.json())
  .then(data => {
    const allResults = [];
    let results = data.results;
    allResults.push(...results);

    // Note: In production, you'd paginate through all pages
    // For now, we'll use the fetched data from services.json if available

    return allResults.map(service => ({
      id: service.id,
      slug: service.slug,
      canonical_id: service.canonical_id,
      name: service.name,
      description: service.description,
      external_url: service.app_url,
      learn_more_url: service.learn_more_url,
      image: service.image,
      primary_color: service.primary_color,
      categories: service.categories,
      popularity: service.popularity,
      is_featured: service.is_featured,
      is_premium: service.is_premium,
      zap_usage_count: service.zap_usage_count,
      hashtag: service.hashtag,
      reads: service.reads || [],
      writes: service.writes || [],
      searches: service.searches || [],
      $id: `https://graph.org.ai/integrations/${service.slug}`
    }));
  });
